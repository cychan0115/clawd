## 1. 核心问题与解决方案

### 1.1 怎么读社区列表（shequ list）

**方案：通过 K8S namespace 获取**

你的每个社区就是一个 K8S namespace：

```bash
# 在 ocr1 集群上执行
ssh ocr1 "kubectl get namespaces | grep -E '^[a-z]+[0-9]*'"
```

预期输出：
```
huadu        Active   45d
jyh          Active   300d
jyh003       Active   120d
jyh011       Active   60d
```

**API 实现：**
```python
# backend/api/community.py
import subprocess
import re

def get_communities():
    """获取所有社区列表"""
    cmd = ['ssh', 'ocr1', 'kubectl get namespaces --no-headers']
    result = subprocess.run(cmd, capture_output=True, text=True)
    
    communities = []
    for line in result.stdout.strip().split('\n'):
        # 匹配社区命名格式：纯小写字母+数字
        match = re.match(r'^([a-z]+[0-9]*)\s+Active', line)
        if match:
            communities.append({
                'id': match.group(1),
                'name': match.group(1),  # 可后续加显示名称映射
                'status': 'Active'
            })
    
    return communities
```

**前端展示：**

```json
    }

```

**如果需要查集群健康状态：**
```bash
# 检查集群是否可达
ssh ocr1 "kubectl cluster-info"

# 输出：
# Kubernetes control plane is running at https://10.0.0.1:6443
```

**API 封装：**
```python
# backend/api/cluster.py
import subprocess

CLUSTER_CONFIG = {
    'ocr1': {'ssh_host': 'ocr1', 'name': 'OCR1 主集群'}
}

def check_cluster_health(cluster_id='ocr1'):
    """检查集群健康状态"""
    host = CLUSTER_CONFIG[cluster_id]['ssh_host']
    try:
        result = subprocess.run(
            ['ssh', host, 'kubectl cluster-info'],
            capture_output=True, text=True, timeout=10
        )
        return {
            'id': cluster_id,
            'status': 'healthy' if result.returncode == 0 else 'unhealthy',
            'message': result.stdout.strip() if result.returncode == 0 else result.stderr.strip()
        }
    except Exception as e:
        return {'id': cluster_id, 'status': 'error', 'message': str(e)}
```

### 1.3 怎么查某个社区具体的队列情况

**这是最核心的部分。你有两种方式：**

#### 方式 A：直接连 MySQL 查（推荐，最快）

**步骤：**
1. SSH 到 ocr1
2. 找到社区对应的 pod（需要读取 pod 的 env 获取 DB 连接信息）
3. 直连 MySQL 执行查询

```python
# backend/api/queue.py
import subprocess
import json
import re
from dataclasses import dataclass

@dataclass
class QueueInfo:
    name: str
    status: int
    count: int

class QueueInspector:
    def __init__(self, cluster='ocr1'):
        self.cluster = cluster
        self.ssh_host = 'ocr1'
    
    def _ssh_exec(self, cmd):
        """在 ocr1 上执行命令"""
        full_cmd = ['ssh', self.ssh_host] + cmd.split()
        result = subprocess.run(full_cmd, capture_output=True, text=True)
        if result.returncode != 0:
            raise Exception(f"SSH command failed: {result.stderr}")
        return result.stdout
    
    def get_db_credentials(self, namespace):
        """从 pod env 中获取数据库连接信息"""
        # 1. 找到主 pod
        pod_cmd = f"kubectl get pods -n {namespace} --selector=app={namespace} -o jsonpath='{{.items[0].metadata.name}}'"
        pod_name = self._ssh_exec(pod_cmd).strip()
        
        # 2. 读取环境变量
        env_cmd = f"kubectl exec -n {namespace} {pod_name} -- env"
        env_output = self._ssh_exec(env_cmd)
        
        # 3. 解析 DB 连接信息
        creds = {}
        for line in env_output.split('\n'):
            if 'DB_HOST' in line or 'MYSQL_HOST' in line:
                creds['host'] = line.split('=')[1]
            elif 'DB_PORT' in line or 'MYSQL_PORT' in line:
                creds['port'] = line.split('=')[1]
            elif 'DB_NAME' in line or 'MYSQL_DATABASE' in line:
                creds['database'] = line.split('=')[1]
            elif 'DB_USER' in line or 'MYSQL_USER' in line:
                creds['user'] = line.split('=')[1]
            elif 'DB_PASSWORD' in line or 'MYSQL_PASSWORD' in line:
                creds['password'] = line.split('=')[1]
        
        return creds
    
    def query_queue_overview(self, namespace):
        """查队列总览 - 按状态分组"""
        creds = self.get_db_credentials(namespace)
        
        sql = """
        SELECT status, COUNT(*) as count 
        FROM queue 
        GROUP BY status 
        ORDER BY status;
        """
        
        # 通过 kubectl exec mysql client 执行
        cmd = f"""kubectl exec -n {namespace} $(kubectl get pods -n {namespace} --selector=app={namespace} -o jsonpath='{{.items[0].metadata.name}}') -- \
        mysql -h{creds['host']} -P{creds['port']} -u{creds['user']} -p{creds['password']} {creds['database']} -e \"{sql}\""""
        
        output = self._ssh_exec(cmd)
        
        # 解析结果
        status_map = {0: 'pending', 1: 'processing', 2: 'done', 3: 'failed'}
        result = {status: 0 for status in status_map.values()}
        
        lines = output.strip().split('\n')[1:]  # 跳过表头
        for line in lines:
            if line.strip():
                parts = line.split()
                status_code = int(parts[0])
                count = int(parts[1])
                result[status_map.get(status_code, 'unknown')] = count
        
        return result
    
    def query_queue_detail(self, namespace):
        """查队列详情 - 按状态和名称分组"""
        creds = self.get_db_credentials(namespace)
        
        sql = """
        SELECT status, name, COUNT(*) as count 
        FROM queue 
        GROUP BY status, name 
        ORDER BY status, count DESC;
        """

        """查报告打印队列详情"""
        creds = self.get_db_credentials(namespace)
        
        sql = """
        SELECT 
            id,
            name,
            status,
            created_at,
            updated_at,
            CASE 
                WHEN name LIKE '%report%' OR name LIKE '%print%' THEN 'report'
```

#### 方式 B：通过 Python 脚本执行（复用现有代码）

你已经有一个 `scripts/inspect_queue.py`，可以直接调用：

```python
# backend/api/queue.py
import subprocess
import json

def get_queue_status_via_script(namespace):
    """调用现有脚本获取队列状态"""
    result = subprocess.run(
        ['python3', 'scripts/inspect_queue.py', namespace],
        capture_output=True, text=True, cwd='/path/to/scripts'
    )
    
    # 解析脚本输出
    output = result.stdout
    
    # 你的脚本输出格式：
    # === huadu 队列 ===
    # 待处理: 11
    # 处理中: 0
    # 已完成: 64372
    # 异常: 9338
    # 总计: 73721
    
    status = {}
    for line in output.split('\n'):
        if '待处理' in line:
            status['pending'] = int(line.split(':')[1].strip())
        elif '处理中' in line:
            status['processing'] = int(line.split(':')[1].strip())
        elif '已完成' in line:
            status['done'] = int(line.split(':')[1].strip())
        elif '异常' in line:
            status['failed'] = int(line.split(':')[1].strip())
    
    return status
```

**优点**：复用已有代码，改动最小
**缺点**：每查一次都要 SSH + 查 env + 连 DB，比直连慢

```vue
<script setup>
import { ref, onMounted, onUnmounted } from 'vue';

const communities = ref([]);
const selectedCommunity = ref('');

const stats = ref([]);
const queueDetail = ref([]);
const loading = ref(false);
const lastUpdated = ref('-');
let refreshTimer = null;

// 加载社区列表
async function loadCommunities() {
  const res = await fetch('/api/communities');

  loading.value = true;

      fetch(`/api/communities/${selectedCommunity.value}/queues/overview`),
      fetch(`/api/communities/${selectedCommunity.value}/queues/detail`)

    const detail = await detailRes.json();

  } finally {
    loading.value = false;

}
```

## 4. 关键技术难点与解决方案

### 4.1 SSH 连接稳定性

**问题**：每次查询都 SSH 到 ocr1，连接可能有延迟或超时

**方案 A：SSH 连接复用**
```bash
# 在 ~/.ssh/config 中配置连接复用
Host ocr1
    HostName ocr1-ip-or-domain
    User your-user
    ControlMaster auto
    ControlPath ~/.ssh/controlmasters/%r@%h:%p
    ControlPersist 10m
```

**方案 B：本地缓存 + 异步刷新**
```python
# 后端缓存策略
from functools import lru_cache
import time

cache = {}
CACHE_TTL = 30  # 30秒

def get_cached_queue_data(namespace):
    """带缓存的队列查询"""
    cache_key = f"queue_{namespace}"
    now = time.time()
    
    if cache_key in cache and now - cache[cache_key]['timestamp'] < CACHE_TTL:
        return cache[cache_key]['data']
    
    # 缓存失效，重新查询
    data = query_queue_overview(namespace)
    cache[cache_key] = {'data': data, 'timestamp': now}
    return data
```

**方案 C（推荐）：用 K8S API 替代 SSH**

如果有 kubeconfig 可以直接访问 ocr1：
```python
from kubernetes import client, config
```

## 6. 快速启动脚本

如果你想快速验证，这里有一个最小可行版本：

```python
# quick_check.py - 本地快速测试
#!/usr/bin/env python3
import subprocess
import sys

def check_queue(community):
    """快速查询社区队列"""
    print(f"\n=== {community} 队列 ===")
    
    # 1. 获取 DB 凭证（从 pod env）
    pod_cmd = f"kubectl get pods -n {community} --selector=app={community} -o jsonpath='{{.items[0].metadata.name}}'"
    pod = subprocess.check_output(['ssh', 'ocr1', pod_cmd]).decode().strip()
    
    env = subprocess.check_output(['ssh', 'ocr1', f'kubectl exec -n {community} {pod} -- env']).decode()
    
    # 解析 DB 连接
    db_vars = {}
    for line in env.split('\n'):
        if '=' in line:
            k, v = line.split('=', 1)
            if 'DB_' in k or 'MYSQL_' in k:
                db_vars[k] = v
    
    # 2. 执行 SQL
    host = db_vars.get('DB_HOST', db_vars.get('MYSQL_HOST', 'localhost'))
    port = db_vars.get('DB_PORT', db_vars.get('MYSQL_PORT', '3306'))
    user = db_vars.get('DB_USER', db_vars.get('MYSQL_USER'))
    password = db_vars.get('DB_PASSWORD', db_vars.get('MYSQL_PASSWORD'))
    database = db_vars.get('DB_NAME', db_vars.get('MYSQL_DATABASE'))
    
    sql = "SELECT status, COUNT(*) FROM queue GROUP BY status;"
    
    mysql_cmd = f"mysql -h{host} -P{port} -u{user} -p{password} {database} -e '{sql}'"
    full_cmd = f"kubectl exec -n {community} {pod} -- {mysql_cmd}"
    
    result = subprocess.check_output(['ssh', 'ocr1', full_cmd]).decode()
    
    # 3. 输出结果
    status_map = {0: '待处理', 1: '处理中', 2: '已完成', 3: '异常'}
    for line in result.split('\n')[1:]:
        if line.strip():
            parts = line.split()
            status = int(parts[0])
            count = parts[1]
            print(f"{status_map.get(status, '未知')}: {count}")

if __name__ == '__main__':
    if len(sys.argv) < 2:
        print("Usage: python3 quick_check.py <community>")
        print("Example: python3 quick_check.py huadu")
        sys.exit(1)
    
    check_queue(sys.argv[1])
```
